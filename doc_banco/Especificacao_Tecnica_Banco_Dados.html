   1. TABELAS DE SEGURANÇA E ACESSO
   ================================================================================== */

-- Tabela:  TB_LOGIN
CREATE TABLE TB_LOGIN (
    CO_SEQ_LOGIN    INT IDENTITY(1,1) NOT NULL,
    NO_USUARIO      VARCHAR(100) NOT NULL,
    DS_EMAIL        VARCHAR(50) NOT NULL,
    CO_LOGIN        VARCHAR(30) NOT NULL,
    TX_SENHA        VARCHAR(100) NOT NULL, 
    NO_PERFIL       VARCHAR(20) NOT NULL,
    ST_INATIVO      BIT DEFAULT 0,
    DH_S_CRIACAO    DATETIME DEFAULT GETDATE(),
    CONSTRAINT PK_LOGIN PRIMARY KEY (CO_SEQ_LOGIN),
    CONSTRAINT UC_LOGIN_EMAIL UNIQUE (DS_EMAIL),
    CONSTRAINT UC_LOGIN_LOGIN UNIQUE (CO_LOGIN)
);
GO

    CO_SEQ_RUMOR        INT IDENTITY(1,1) NOT NULL,
    CO_IDU              VARCHAR(20),
    NO_TITULO           VARCHAR(200) NOT NULL,
    TX_DESCRICAO        VARCHAR(500),
    ST_VERACIDADE       VARCHAR(10),
    DS_FUNDAMENTO_VERACIDADE VARCHAR(300),
    NO_NOTIFICADOR_FONTE VARCHAR(100),
    DT_RECEBIMENTO      DATE,
    NO_TIPO_VIGILANCIA  VARCHAR(10),
    DS_TIPO_ENCAMINHAMENTO VARCHAR(30),
    DS_NIVEL_CONFIANCA  VARCHAR(50),
    
    -- Foreign Keys Colunas
    CO_SEQ_STATUS       INT,
    CO_SEQ_NATUREZA     INT,
    CO_SEQ_ICMRA        INT,
    CO_SEQ_PAIS         INT,
    CO_SEQ_ESTADO       INT,
    CO_SEQ_CIDADE       INT,
    CO_SEQ_USUARIO_CADASTRO INT,
    CO_SEQ_LOCAL_EVENTO INT,
    DH_S_CADASTRO       DATETIME,
    
    CONSTRAINT PK_RUMOR PRIMARY KEY (CO_SEQ_RUMOR)
);
GO
